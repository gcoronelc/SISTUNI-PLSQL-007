CREATE OR REPLACE FUNCTION FN_CALCULAR_MCD
(P_PA NUMBER, P_PB NUMBER) 
RETURN NUMBER
AS
	V_RES NUMBER;
	V_MAYOR NUMBER;
	V_MENOR NUMBER;
BEGIN 
	IF(P_PA > P_PB) THEN
		V_MAYOR := P_PA;
		V_MENOR := P_PB;
	ELSE
		V_MAYOR := P_PB;
		V_MENOR := P_PA; 
	END IF;
	
	 WHILE V_MENOR != 0 LOOP
		V_RES:= V_MENOR;
		V_MENOR :=  MOD(V_MAYOR,V_MENOR);	    	    
	    V_MAYOR := V_RES;
  	END LOOP;	
	
	RETURN V_RES;
END;

-- LLAMAR FUNCION
DECLARE
V_RESULT NUMBER;
BEGIN
V_RESULT := FN_CALCULAR_MCD(9,27);
DBMS_OUTPUT.PUT_LINE('RSULTADOS: ' || V_RESULT);
END;
