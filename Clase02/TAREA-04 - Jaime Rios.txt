
CREATE OR REPLACE FUNCTION SCOTT.FN_CONDICION
(P_NOTA NUMBER)
RETURN VARCHAR2 AS
 V_CONDICION VARCHAR2(200);
BEGIN
 IF (P_NOTA<0 OR P_NOTA >20) THEN
 	V_CONDICION := 'NOTA NO VALIDA';
 ELSE
	 CASE 
	 	WHEN P_NOTA >=14 THEN V_CONDICION := 'APROBADO';
	 	WHEN (P_NOTA >=10 AND P_NOTA <14) THEN V_CONDICION := 'ASISTENTE';
	 	ELSE V_CONDICION := 'DESAPROBADO';
	 END CASE;
 END IF;
 RETURN V_CONDICION;
END;

SELECT SCOTT.FN_CONDICION(22) FROM DUAL;
