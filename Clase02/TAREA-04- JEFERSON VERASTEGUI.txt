CREATE OR REPLACE FUNCTION SCOTT.FN_CONDICION
(P_NUM NUMBER)
RETURN VARCHAR2
AS
V_LETRAS VARCHAR2(200);
BEGIN
CASE 
WHEN P_NUM>=14.0 THEN V_LETRAS:='APROBADO';
WHEN P_NUM>=10 AND P_NUM<14.0 THEN V_LETRAS:='ASISTENTE';
WHEN P_NUM<10 THEN V_LETRAS:='DESAPROBADO';
END CASE;
RETURN V_LETRAS;
END;

SELECT SCOTT.FN_CONDICION(9) FROM DUAL;
