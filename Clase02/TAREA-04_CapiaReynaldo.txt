CREATE OR REPLACE FUNCTION SCOTT.FN_NOTA_ALUMNOS
( P_NOTA NUMBER )
RETURN VARCHAR2
AS
  V_MENSAJE VARCHAR2(200);
BEGIN
  CASE 
    WHEN P_NOTA <=20 AND P_NOTA >= 14 THEN V_MENSAJE := 'APROBADO';
    WHEN P_NOTA <=13 AND P_NOTA >= 10  THEN V_MENSAJE := 'ASISTENTE';
    WHEN P_NOTA < 10 AND P_NOTA >=0 THEN V_MENSAJE := 'DESAPROBADO';
    ELSE V_MENSAJE := 'NOTA NO EXISTENTE';
   END CASE;
   RETURN V_MENSAJE;
END;
/

SELECT SCOTT.FN_NOTA_ALUMNOS(15) FROM DUAL;
