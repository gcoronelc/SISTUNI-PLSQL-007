
CREATE OR REPLACE PROCEDURE RECURSOS.SP_TAREA3
( P_IDEMPLEADO IN CHAR, P_VALOR1 OUT NUMBER)
AS
	V_COUNT NUMBER := 0;
BEGIN
	SELECT count(1) INTO V_COUNT 
	FROM EMPLEADO 
	WHERE IDEMPLEADO = P_IDEMPLEADO;
	
	IF V_COUNT = 1 THEN
		select sueldo INTO P_VALOR1 
		from EMPLEADO 
		WHERE IDEMPLEADO = P_IDEMPLEADO;
	ELSE
		P_VALOR1 := -1;  
	END IF;
END;
/

DECLARE 
  V_VALOR1 number;
BEGIN
  RECURSOS.SP_TAREA3 ('E0001', V_VALOR1);
  DBMS_OUTPUT.PUT_LINE('1: ' || V_VALOR1);
END;
/
