CREATE OR REPLACE PROCEDURE RECURSOS.MOSTRAR_SUELDO2 (id_llamar  VARCHAR2)
as
SUELDO_1 numeric;
SUELDO_2 numeric;
begin
SELECT SUELDO INTO SUELDO_1 
FROM RECURSOS.EMPLEADO A
WHERE A.idempleado=id_llamar;
SELECT NVL(SUELDO_1,-1) INTO SUELDO_2 FROM DUAL;
DBMS_OUTPUT.PUT_LINE(SUELDO_2);
 END;
/

call RECURSOS.MOSTRAR_SUELDO2('E9901');
